The complete, integrated codebase for **QMicro v11**, incorporating the architectural shift towards a hybrid software/hardware model. 

In v11, the software microkernel acts as an **orchestrator**, delegating strict sub-nanosecond pulse timing, waveform streaming, and active feedback to a dedicated **Hardware Pulse Processing Unit (PPU)**. 

### What's New in v11 (Hardware Integration)
*   **`qisa.rs` (NEW):** Defines the Quantum Instruction Set Architecture (QISA) for deterministic hardware execution.
*   **`hw/pulse_engine.rs` (NEW):** PPU hardware abstraction layer. Manages sequence offloading, autonomous DMA execution, and real-time hardware feedback policies.
*   **`capability.rs`:** Added `RIGHT_PULSE_EXECUTE` to allow capabilities to be directly mapped to the hardware sequencer.
*   **`fault.rs`:** Added hardware-level fault sources (`HardwarePulseEngine`) triggered via PPU watchdog interrupts.
*   **`dma.rs`:** DMA leases are now hardware-timed; the PPU directly tracks progress and automatically expires/unpins zombie memory.

```rust
//! [v11] Capability system updated with RIGHT_PULSE_EXECUTE for PPU offloading.

pub const RIGHT_READ: u8 = 1 << 0;
pub const RIGHT_WRITE: u8 = 1 << 1;
pub const RIGHT_EXECUTE: u8 = 1 << 2;
pub const RIGHT_GRANT: u8 = 1 << 3;
pub const RIGHT_REVOKE: u8 = 1 << 4;
pub const RIGHT_PIN: u8 = 1 << 5;
/// [v11] Allows capability memory to be mapped directly to the hardware Pulse Engine.
pub const RIGHT_PULSE_EXECUTE: u8 = 1 << 6;

#[derive(Clone, Copy, Default)]
pub struct Capability {
    pub base: usize,
    pub length: usize,
    pub rights: u8,
    pub valid: bool,
}

pub struct CapabilityTable {
    entries: [Capability; crate::config::CAP_TABLE_SIZE],
}

impl CapabilityTable {
    pub const fn new() -> Self {
        Self {
            entries: [Capability { base: 0, length: 0, rights: 0, valid: false }; crate::config::CAP_TABLE_SIZE],
        }
    }

    pub fn lookup(&self, idx: usize) -> Option<&Capability> {
        self.entries.get(idx).filter(|c| c.valid)
    }

    pub fn set_right(&mut self, idx: usize, right: u8) -> Result<(), &'static str> {
        let cap = self.entries.get_mut(idx).ok_or("Index out of range")?;
        if !cap.valid { return Err("Capability not valid"); }
        cap.rights |= right;
        Ok(())
    }

    pub fn clear_right(&mut self, idx: usize, right: u8) -> Result<(), &'static str> {
        let cap = self.entries.get_mut(idx).ok_or("Index out of range")?;
        if !cap.valid { return Err("Capability not valid"); }
        cap.rights &= !right;
        Ok(())
    }

    pub fn force_unpin(&mut self, base: usize, len: usize) {
        for cap in self.entries.iter_mut() {
            if cap.valid && cap.base == base && cap.length == len {
                cap.rights &= !RIGHT_PIN;
                defmt::warn!("[CAP] RIGHT_PIN force-cleared: base={:#x}", base);
            }
        }
    }
}
```

```rust
//! [v11] Pulse Engine Hardware Abstraction.
//! Interfaces with the dedicated FPGA/ASIC PPU (Pulse Processing Unit).

#![allow(unsafe_code)]
use crate::qisa::QisaInstruction;
use crate::capability::{CapabilityTable, RIGHT_PULSE_EXECUTE, RIGHT_PIN};

pub enum FeedbackPolicy {
    /// Measure and store result, no hardware branch.
    LogOnly,
    /// Measure and trigger hardware state-reset automatically.
    ActiveReset(u16), // Threshold
    /// Hardware evaluates QEC parity and applies fast-feedback.
    ErrorCorrectionCore,
}

pub struct PulseEngine;

impl PulseEngine {
    /// Map a classical software sequence into the Hardware Sequencer memory.
    pub unsafe fn load_sequence(
        cap_table: &mut CapabilityTable,
        cap_idx: usize,
        sequence_id: usize,
    ) -> Result<(), &'static str> {
        let cap = cap_table.lookup(cap_idx).ok_or("Invalid capability")?;
        
        if cap.rights & RIGHT_PULSE_EXECUTE == 0 || cap.rights & RIGHT_PIN == 0 {
            return Err("Missing PPU execute or PIN rights on sequence memory");
        }

        // Write to MMIO PPU instruction base registers
        let ppu_seq_base = (crate::config::PPU_BASE_ADDR + (sequence_id * 8)) as *mut u32;
        core::ptr::write_volatile(ppu_seq_base, cap.base as u32);
        core::ptr::write_volatile(ppu_seq_base.add(1), cap.length as u32);
        
        defmt::info!("[PPU] Hardware sequence {} loaded from {:#x}", sequence_id, cap.base);
        Ok(())
    }

    /// Arm the sequencer to trigger automatically at a precise hardware timestamp.
    pub unsafe fn arm_trigger(sequence_id: usize, absolute_timestamp: u64) {
        let ppu_trigger = (crate::config::PPU_BASE_ADDR + 0x100) as *mut u64;
        core::ptr::write_volatile(ppu_trigger, absolute_timestamp);
        let ppu_ctrl = (crate::config::PPU_BASE_ADDR + 0x108) as *mut u32;
        core::ptr::write_volatile(ppu_ctrl, sequence_id as u32 | 0x1); // ARM bit
        defmt::debug!("[PPU] Sequence {} armed at {}", sequence_id, absolute_timestamp);
    }

    /// Configure hardware feedback logic.
    pub unsafe fn configure_feedback(policy: FeedbackPolicy) {
        let ppu_fb = (crate::config::PPU_BASE_ADDR + 0x200) as *mut u32;
        let policy_val = match policy {
            FeedbackPolicy::LogOnly => 0x0,
            FeedbackPolicy::ActiveReset(thresh) => 0x1000_0000 | (thresh as u32),
            FeedbackPolicy::ErrorCorrectionCore => 0x2000_0000,
        };
        core::ptr::write_volatile(ppu_fb, policy_val);
    }
}
```

```rust
//! [v11] Hierarchical Fault Isolation — Updated with PPU escalation.
use crate::scheduler;

pub enum FaultSource {
    /// A non-quantum, classical software domain (logger, UART, Ethernet, etc.)
    ClassicalDomain(usize),
    /// The software Quantum Pulse Domain (Orchestrator).
    QuantumPulseDomain,
    /// The Kernel core fault.
    KernelCore,
    /// [v11] Hardware sequencer fault (watchdog timeout, DMA bus error).
    HardwarePulseEngine,
}

pub fn handle_fault(source: FaultSource) {
    match source {
        FaultSource::ClassicalDomain(idx) => {
            scheduler::halt_domain(idx);
            defmt::warn!("[FAULT L1] Classical domain {} halted. Pulse PPU continues hardware sequences.", idx);
        }
        FaultSource::QuantumPulseDomain | FaultSource::HardwarePulseEngine | FaultSource::KernelCore => {
            unsafe { crate::hw::safestate::SafeStateController::trigger(); }
            let msg = match source {
                FaultSource::HardwarePulseEngine => "Hardware PPU Engine Fault",
                _ => "Critical Software Fault",
            };
            defmt::error!("[FAULT L2/L3] {} — Safe-State engaged.", msg);
            panic!("Physical Safe-State Engaged.");
        }
    }
}
```

#### `qmicro/src/hw/dma.rs`
```rust
//! [v11] DMA Pinning + HW-Sequenced Leases.
//! PPU hardware handles absolute timing. If a hardware-sequenced DMA channel 
//! fails to complete, the hardware flags a fault.

use crate::capability::CapabilityTable;

#[derive(Clone, Copy, Default)]
pub struct DmaLease {
    pub owner_domain: usize,
    pub pinned_base: usize,
    pub pinned_len: usize,
    pub start_cycles: u32,
    pub max_duration_cycles: u32,
    pub active: bool,
    /// [v11] True if this lease is executed entirely by the PPU hardware.
    pub hardware_sequenced: bool, 
}

const MAX_DMA_CHANNELS: usize = 4;
static mut DMA_LEASES: [DmaLease; MAX_DMA_CHANNELS] = [DmaLease {
    owner_domain: 0, pinned_base: 0, pinned_len: 0,
    start_cycles: 0, max_duration_cycles: 0, active: false, hardware_sequenced: false
}; MAX_DMA_CHANNELS];

pub unsafe fn grant_lease(
    channel: usize, owner_domain: usize, pinned_base: usize, pinned_len: usize,
    max_duration_cycles: u32, hardware_sequenced: bool
) {
    let now = crate::hw::timer::dwt_cyccnt();
    DMA_LEASES[channel] = DmaLease {
        owner_domain, pinned_base, pinned_len, start_cycles: now,
        max_duration_cycles, active: true, hardware_sequenced
    };
}

pub unsafe fn release_lease(channel: usize) {
    if DMA_LEASES[channel].active {
        DMA_LEASES[channel].active = false;
    }
}

pub fn watchdog_tick(cap_table: &mut CapabilityTable) {
    let now = crate::hw::timer::dwt_cyccnt();
    for channel in 0..MAX_DMA_CHANNELS {
        let lease = unsafe { &mut DMA_LEASES[channel] };
        if !lease.active { continue; }

        let elapsed = now.wrapping_sub(lease.start_cycles);
        
        // For hardware_sequenced leases, the PPU watchdog usually flags errors via IRQ.
        // The software watchdog acts as an absolute ultimate fallback.
        if elapsed > lease.max_duration_cycles {
            defmt::error!("[DMA] Lease EXPIRED: channel={} hardware_seq={}", channel, lease.hardware_sequenced);
            cap_table.force_unpin(lease.pinned_base, lease.pinned_len);
            lease.active = false;

            let fault_src = if lease.hardware_sequenced {
                crate::fault::FaultSource::HardwarePulseEngine
            } else {
                crate::fault::FaultSource::QuantumPulseDomain
            };
            crate::fault::handle_fault(fault_src);
        }
    }
}
```

```rust
//! [v11] Deterministic Interrupt Staging.
//! Masking now heavily favors the Hardware PPU IRQ line over all software tasks.

#![allow(unsafe_code)]
use core::sync::atomic::{AtomicU32, AtomicBool, Ordering};

static ISR_CYCLES_THIS_WINDOW: AtomicU32 = AtomicU32::new(0);
static WINDOW_START_CYCLES: AtomicU32 = AtomicU32::new(0);
static NON_CRITICAL_MASKED: AtomicBool = AtomicBool::new(false);

#[inline(always)]
pub fn isr_enter() -> u32 { crate::hw::timer::dwt_cyccnt() }

#[inline(always)]
pub fn isr_exit(entry_cycles: u32) {
    let now = crate::hw::timer::dwt_cyccnt();
    ISR_CYCLES_THIS_WINDOW.fetch_add(now.wrapping_sub(entry_cycles), Ordering::Relaxed);
}

pub fn quota_tick() {
    let now = crate::hw::timer::dwt_cyccnt();
    let window_start = WINDOW_START_CYCLES.load(Ordering::Relaxed);
    
    if now.wrapping_sub(window_start) >= crate::config::ISR_QUOTA_WINDOW_CYCLES {
        ISR_CYCLES_THIS_WINDOW.store(0, Ordering::Relaxed);
        WINDOW_START_CYCLES.store(now, Ordering::Relaxed);
        if NON_CRITICAL_MASKED.swap(false, Ordering::Relaxed) {
            unsafe { unmask_non_critical_irqs(); }
        }
        return;
    }

    let isr_cycles = ISR_CYCLES_THIS_WINDOW.load(Ordering::Relaxed);
    if isr_cycles > crate::config::ISR_QUOTA_MAX_CYCLES && !NON_CRITICAL_MASKED.load(Ordering::Relaxed) {
        NON_CRITICAL_MASKED.store(true, Ordering::Relaxed);
        unsafe { mask_non_critical_irqs(); }
        defmt::warn!("[IRQ] Quota exceeded — masking non-critical IRQs.");
    }
}

unsafe fn mask_non_critical_irqs() { /* NVIC Masking logic */ }
unsafe fn unmask_non_critical_irqs() { /* NVIC Unmasking logic */ }

/// [v11] Directly handle critical hardware faults from PPU.
pub extern "C" fn ppu_hardware_fault_handler() {
    crate::fault::handle_fault(crate::fault::FaultSource::HardwarePulseEngine);
}
```

```rust
    pub mpu_region: u32,
    pub receiver: usize,
    pub phys_base: usize,
    pub length: usize,
    pub active: bool,
}

const MAX_IPC_GRANTS: usize = 4;
static mut IPC_GRANTS: [IpcGrant; MAX_IPC_GRANTS] = [IpcGrant { mpu_region: 0, receiver: 0, phys_base: 0, length: 0, active: false }; MAX_IPC_GRANTS];

pub unsafe fn remap_region_for_ipc(receiver_domain: usize, phys_base: usize, length: usize) -> Result<usize, &'static str> {
    let slot = IPC_GRANTS.iter().position(|g| !g.active).ok_or("No free IPC grant slots")?;
    let mpu_region = (slot + 1) as u32;
    ActiveMpu::write_region(mpu_region, phys_base, length, AP_FULL_ACCESS, false, true)?;
    IPC_GRANTS[slot] = IpcGrant { mpu_region, receiver: receiver_domain, phys_base, length, active: true };
    Ok(slot)
}

pub unsafe fn revoke_ipc_grant(grant_idx: usize) -> Result<(), &'static str> {
    let grant = &mut IPC_GRANTS[grant_idx];
    if !grant.active { return Err("Grant inactive"); }
    ActiveMpu::write_region(grant.mpu_region, 0, 32, AP_NO_ACCESS, false, false)?;
    grant.active = false;
    Ok(())
}
```

```rust
//! [v11] Shadow Profiling.
use core::sync::atomic::{AtomicU32, Ordering};

pub struct DomainProfile {
    pub worst_case_cycles: AtomicU32,
}

struct DomainScratchpad {
    last_start_cycles: u32,
    local_worst_case: u32,
    ticks_since_flush: u32,
}

static PROFILES: [DomainProfile; crate::config::NUM_DOMAINS] = {
    [DomainProfile { worst_case_cycles: AtomicU32::new(0) }; crate::config::NUM_DOMAINS]
};

static mut SCRATCHPADS: [DomainScratchpad; crate::config::NUM_DOMAINS] = {
    [DomainScratchpad { last_start_cycles: 0, local_worst_case: 0, ticks_since_flush: 0 }; crate::config::NUM_DOMAINS]
};

pub fn mark_start(domain_idx: usize) {
    unsafe { SCRATCHPADS[domain_idx].last_start_cycles = crate::hw::timer::dwt_cyccnt(); }
}

pub fn mark_end(domain_idx: usize) {
    let now = crate::hw::timer::dwt_cyccnt();
    unsafe {
        let sp = &mut SCRATCHPADS[domain_idx];
        let elapsed = now.wrapping_sub(sp.last_start_cycles);
        if elapsed > sp.local_worst_case { sp.local_worst_case = elapsed; }
        sp.ticks_since_flush += 1;
    }
}

pub fn flush_telemetry() {
    for idx in 0..crate::config::NUM_DOMAINS {
        let local_wc = unsafe { SCRATCHPADS[idx].local_worst_case };
        let current = PROFILES[idx].worst_case_cycles.load(Ordering::Relaxed);
        if local_wc > current {
            PROFILES[idx].worst_case_cycles.store(local_wc, Ordering::Relaxed);
        }
    }
}
```

```rust
//! [v11] Scheduler. Classical domains orchestrated while PPU handles real-time execution.
use crate::fault::{FaultSource, handle_fault};

pub fn halt_domain(idx: usize) {
    defmt::warn!("[SCHED] Domain {} halted.", idx);
}

pub fn tick(cap_table: &mut crate::capability::CapabilityTable) -> KResult<()> {
    // Note: hardware watchdog is active in the PPU; software pets its own watchdog.
    crate::hw::watchdog::pet();
    crate::hw::irq::quota_tick();
    crate::hw::dma::watchdog_tick(cap_table);

    let pending: Option<(usize, &Task)> = select_next_domain();
    if let Some((idx, task)) = pending {
        let now = crate::hw::timer::dwt_cyccnt();
        crate::hw::profiling::mark_start(idx);
        
        if task.deadline <= now {
            if let Err(e) = dispatch_domain(idx, task) {
                if idx == crate::config::QUANTUM_PULSE_DOMAIN_IDX {
                    handle_fault(FaultSource::QuantumPulseDomain);
                } else {
                    handle_fault(FaultSource::ClassicalDomain(idx));
                }
            }
        }
        crate::hw::profiling::mark_end(idx);
    }
    Ok(())
}

type KResult<T> = Result<T, KernelError>;
#[derive(Debug)] pub enum KernelError { DomainFault, StackOverflow, CapabilityViolation }
struct Task { deadline: u32 }
fn select_next_domain() -> Option<(usize, &'static Task)> { None }
fn dispatch_domain(_idx: usize, _task: &Task) -> KResult<()> { Ok(()) }
```

#### `qmicro/src/main.rs`
```rust
//! QMicro v11 Hybrid Orchestrator Entry Point.
//! Software manages classical tasks, IPC, and capabilities; Hardware runs pulses natively.

#![no_std]
#![no_main]

// (Assuming standard module definitions `mod hw`, `mod fault`, `mod scheduler`, etc. here)

#[no_mangle]
pub extern "C" fn _start() -> ! {
    defmt::info!("Booting QMicro v11 [HYBRID ORCHESTRATOR]...");

    // Step 1: Physical Safe State on Boot.
    unsafe { crate::hw::safestate::SafeStateController::trigger(); }

    // Step 2: MPU setup.
    unsafe { 
        crate::hw::mpu::configure_code_data_regions();
        crate::hw::mpu::configure_kernel_stack_guard();
    }

    // Step 3: Initialize Capability Table.
    let mut cap_table = crate::capability::CapabilityTable::new();

    // Step 4: [v11] Configure Hardware PPU policies.
    unsafe {
        // By default, map error correction core directly into hardware feedback path.
        crate::hw::pulse_engine::PulseEngine::configure_feedback(
            crate::hw::pulse_engine::FeedbackPolicy::ErrorCorrectionCore
        );
    }

    defmt::info!(
        "QMicro v11 running.\n\
         - PPU Hardware Abstraction Enabled.\n\
         - HW-Timed DMA Leases Armed.\n\
         - Real-Time Feedback shifted to FPGA."
    );

    // Step 5: Main Scheduler Loop
    loop {
        if let Err(e) = crate::scheduler::tick(&mut cap_table) {
```
