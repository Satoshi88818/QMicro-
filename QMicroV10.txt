//! [IMP-v10-1] Hierarchical Fault Isolation — Fault Escalation Matrix. //! //! Replaces the flat global SafeState trigger with a three-level escalation: //! Level 1 — Classical domain fault → halt that domain only //! Level 2 — Quantum Pulse fault → trigger SafeState //! Level 3 — Kernel core fault → SafeState + panic use crate::scheduler; /// Identifies the origin of a fault so the escalation matrix can respond /// with the minimum necessary disruption. pub enum FaultSource { /// A non-quantum, classical software domain (logger, UART, Ethernet, etc.) ClassicalDomain(usize), /// The Quantum Pulse Generator domain — physical qubit control is affected. QuantumPulseDomain, /// The kernel itself (scheduler, MPU, capability table). KernelCore, } /// Entry point for all hardware and software faults. /// /// Call this instead of directly invoking `SafeStateController::trigger()` /// so that the minimum disruption policy is applied automatically. pub fn handle_fault(source: FaultSource) { match source { // ── Level 1: Soft fault in a classical domain ────────────────────── FaultSource::ClassicalDomain(idx) => { scheduler::halt_domain(idx); defmt::warn!( "[FAULT L1] Classical domain {} halted. Pulse-lock maintained.", idx ); // The quantum pulse domain continues uninterrupted. } // ── Level 2/3: Quantum or kernel fault — physical safety first ───── FaultSource::QuantumPulseDomain | FaultSource::KernelCore => { // Immediately drive QPU control lines to safe voltages. unsafe { crate::hw::safestate::SafeStateController::trigger(); } defmt::error!("[FAULT L2/L3] Critical fault — Safe-State engaged."); panic!("Critical Fault: Physical Safe-State Engaged."); } } }

qmicro/src/hw/profiling.rs (v10 — Shadow Profiling)

//! [IMP-v10-2] Shadow Profiling — Lock-Free Per-Domain Scratchpads. //! //! v9 used AtomicU32::compare_exchange in mark_end, which introduced MESI //! cache-coherency overhead on the high-priority dispatch path. //! //! v10 gives each domain a private, non-atomic `local_worst_case` that only //! the scheduler writes. A low-priority Telemetry Domain periodically reads //! these scratchpads and aggregates them into the global PROFILES atomics for //! external consumption (ITM, diagnostics), keeping the hot path atomic-free. use core::sync::atomic::{AtomicU32, Ordering}; /// Public profile entry — used by the Telemetry Domain for reporting. pub struct DomainProfile { /// Worst-case execution time accumulated by the Telemetry Domain. pub worst_case_cycles: AtomicU32, } /// Private per-domain scratchpad — only ever accessed by the scheduler. /// No atomics needed because the scheduler serialises domain dispatch. struct DomainScratchpad { last_start_cycles: u32, local_worst_case: u32, ticks_since_flush: u32, } // ── Static storage ──────────────────────────────────────────────────────────── static PROFILES: [DomainProfile; crate::config::NUM_DOMAINS] = { const INIT: DomainProfile = DomainProfile { worst_case_cycles: AtomicU32::new(0), }; [INIT; crate::config::NUM_DOMAINS] }; // Safety: each scratchpad is accessed only by the single-threaded scheduler. static mut SCRATCHPADS: [DomainScratchpad; crate::config::NUM_DOMAINS] = { const INIT: DomainScratchpad = DomainScratchpad { last_start_cycles: 0, local_worst_case: 0, ticks_since_flush: 0, }; [INIT; crate::config::NUM_DOMAINS] }; // ── Hot path (called from the scheduler, zero atomics) ──────────────────────── /// Record the start of a domain's execution slice. /// Must only be called from the scheduler context. pub fn mark_start(domain_idx: usize) { let now = crate::hw::timer::dwt_cyccnt(); unsafe { SCRATCHPADS[domain_idx].last_start_cycles = now; } } /// Record the end of a domain's execution slice and update the local worst-case. /// Must only be called from the scheduler context. pub fn mark_end(domain_idx: usize) { let now = crate::hw::timer::dwt_cyccnt(); unsafe { let sp = &mut SCRATCHPADS[domain_idx]; let elapsed = now.wrapping_sub(sp.last_start_cycles); if elapsed > sp.local_worst_case { sp.local_worst_case = elapsed; } sp.ticks_since_flush += 1; } } // ── Cold path (called by the low-priority Telemetry Domain) ────────────────── /// Aggregate local scratchpads into the global PROFILES. /// Call this from a low-priority Telemetry Domain, not from the scheduler tick. pub fn flush_telemetry() { for idx in 0..crate::config::NUM_DOMAINS { let local_wc = unsafe { SCRATCHPADS[idx].local_worst_case }; // Relaxed store is fine: Telemetry Domain is the sole writer here. let current = PROFILES[idx].worst_case_cycles.load(Ordering::Relaxed); if local_wc > current { PROFILES[idx] .worst_case_cycles .store(local_wc, Ordering::Relaxed); } // Optionally emit over ITM defmt::trace!("[TELEMETRY] Domain {} WCET: {} cycles", idx, local_wc); } } /// Read the aggregated worst-case for a domain (safe to call from any context). pub fn read_worst_case(domain_idx: usize) -> u32 { PROFILES[domain_idx].worst_case_cycles.load(Ordering::Relaxed) }

//! [IMP-v9-4 / IMP-v10-3] DMA Pinning + Lease Watchdog. //! //! v9 introduced RIGHT_PIN to prevent revocation during active DMA. //! v10 adds a lease timer so a hung Pulse Processor (P4) cannot pin memory //! forever ("Zombie Memory"). If the DMA hasn't signalled completion within //! `max_duration_cycles`, the kernel forces SafeState on that channel and //! forcibly unpins the memory. use crate::capability::CapabilityTable; /// Represents an active DMA lease held by a domain. #[derive(Clone, Copy, Default)] pub struct DmaLease { /// The domain that owns this DMA channel. pub owner_domain: usize, /// Physical base address of the pinned memory region. pub pinned_base: usize, /// Length of the pinned region in bytes. pub pinned_len: usize, /// CPU cycle counter value when the lease was granted. pub start_cycles: u32, /// Maximum duration the lease may run before forced expiry. pub max_duration_cycles: u32, /// Whether this lease slot is currently active. pub active: bool, } const MAX_DMA_CHANNELS: usize = 4; static mut DMA_LEASES: [DmaLease; MAX_DMA_CHANNELS] = [DmaLease { owner_domain: 0, pinned_base: 0, pinned_len: 0, start_cycles: 0, max_duration_cycles: 0, active: false, }; MAX_DMA_CHANNELS]; /// Grant a DMA lease for `channel`. /// Sets RIGHT_PIN on the memory capability and records the lease timer. /// /// # Safety /// Must only be called from the kernel/scheduler context. pub unsafe fn grant_lease( channel: usize, owner_domain: usize, pinned_base: usize, pinned_len: usize, max_duration_cycles: u32, ) { debug_assert!(channel < MAX_DMA_CHANNELS); let now = crate::hw::timer::dwt_cyccnt(); DMA_LEASES[channel] = DmaLease { owner_domain, pinned_base, pinned_len, start_cycles: now, max_duration_cycles, active: true, }; defmt::debug!( "[DMA] Lease granted: channel={} domain={} base={:#x} len={} max_cycles={}", channel, owner_domain, pinned_base, pinned_len, max_duration_cycles ); } /// Signal that a DMA transfer has completed and release the lease. /// /// # Safety /// Must only be called from the DMA completion ISR or equivalent. pub unsafe fn release_lease(channel: usize) { debug_assert!(channel < MAX_DMA_CHANNELS); if DMA_LEASES[channel].active { DMA_LEASES[channel].active = false; defmt::debug!("[DMA] Lease released: channel={}", channel); } } /// Called every scheduler tick to check for expired leases. /// /// If a lease has exceeded its `max_duration_cycles`, the kernel forcibly /// unpins the memory and triggers a Level-2 fault via the escalation matrix. pub fn watchdog_tick(cap_table: &mut CapabilityTable) { let now = crate::hw::timer::dwt_cyccnt(); for channel in 0..MAX_DMA_CHANNELS { let lease = unsafe { &mut DMA_LEASES[channel] }; if !lease.active { continue; } let elapsed = now.wrapping_sub(lease.start_cycles); if elapsed > lease.max_duration_cycles { defmt::error!( "[DMA] Lease EXPIRED: channel={} domain={} elapsed={} max={}", channel, lease.owner_domain, elapsed, lease.max_duration_cycles ); // Force-unpin the memory region cap_table.force_unpin(lease.pinned_base, lease.pinned_len); lease.active = false; // Escalate: treat the Pulse domain hanging as a quantum fault crate::fault::handle_fault( crate::fault::FaultSource::QuantumPulseDomain ); } } }

//! [IMP-v10-4] MPU-Backed Zero-Copy IPC public API. //! //! Provides `ipc_grant(target_domain, memory_range)` for the Classical Logic //! domain to hand large pulse waveform tables to the Quantum Driver domain //! with 0ns of copy time. use crate::capability::{CapabilityTable, RIGHT_PIN}; /// Grant `target_domain` read access to `(base, len)` via MPU re-mapping. /// /// The sender's capability must already exist in `cap_table` and will have /// RIGHT_PIN set so it cannot be modified or revoked while the grant is live. /// /// Returns the grant index which must be passed to `ipc_revoke` when done. pub fn ipc_grant( cap_table: &mut CapabilityTable, sender_cap: usize, target_domain: usize, base: usize, len: usize, ) -> Result<usize, &'static str> { // Pin the sender's capability to prevent modification during the grant. cap_table.set_right(sender_cap, RIGHT_PIN)?; // Re-map the physical region into the receiver's MPU view. let grant_idx = unsafe { crate::hw::mpu::remap_region_for_ipc(target_domain, base, len)? }; defmt::info!( "[IPC] Grant issued: cap={} target={} base={:#x} len={} grant_idx={}", sender_cap, target_domain, base, len, grant_idx ); Ok(grant_idx) } /// Revoke an IPC grant and unpin the sender's capability. pub fn ipc_revoke( cap_table: &mut CapabilityTable, sender_cap: usize, grant_idx: usize, ) -> Result<(), &'static str> { unsafe { crate::hw::mpu::revoke_ipc_grant(grant_idx)?; } cap_table.clear_right(sender_cap, RIGHT_PIN)?; defmt::info!("[IPC] Grant revoked: cap={} grant_idx={}", sender_cap, grant_idx); Ok(()) } 

qmicro/src/hw/irq.rs (NEW — v10)

//! [IMP-v10-5] Deterministic Interrupt Staging — Interrupt Quotas. //! //! High-frequency interrupts (e.g. a 100kHz cryo-monitor) can starve the //! scheduler and introduce jitter into qubit pulse timing. //! //! This module tracks how many CPU cycles are spent in ISRs within each //! rolling 1ms window. If ISRs exceed 10% of the window, non-critical //! interrupt lines (UART, Ethernet) are temporarily masked until the window //! resets, protecting the Pulse Timing budget. #![allow(unsafe_code)] use core::sync::atomic::{AtomicU32, AtomicBool, Ordering}; // ── State ───────────────────────────────────────────────────────────────────── /// Cycles spent in ISRs since the start of the current quota window. static ISR_CYCLES_THIS_WINDOW: AtomicU32 = AtomicU32::new(0); /// CPU cycle timestamp at the start of the current quota window. static WINDOW_START_CYCLES: AtomicU32 = AtomicU32::new(0); /// True when non-critical ISRs are currently masked. static NON_CRITICAL_MASKED: AtomicBool = AtomicBool::new(false); // ── ISR instrumentation helpers ─────────────────────────────────────────────── /// Call at the very start of every non-critical ISR to record entry time. /// Returns the entry cycle count (pass to `isr_exit`). #[inline(always)] pub fn isr_enter() -> u32 { crate::hw::timer::dwt_cyccnt() } /// Call at the very end of every non-critical ISR to accumulate ISR cycles. #[inline(always)] pub fn isr_exit(entry_cycles: u32) { let now = crate::hw::timer::dwt_cyccnt(); let elapsed = now.wrapping_sub(entry_cycles); ISR_CYCLES_THIS_WINDOW.fetch_add(elapsed, Ordering::Relaxed); } // ── Quota enforcement (called from scheduler tick) ──────────────────────────── /// Evaluate the ISR quota for the current window. /// /// Must be called once per scheduler tick. Rolls the window when /// `ISR_QUOTA_WINDOW_CYCLES` have elapsed and re-enables masked interrupts. pub fn quota_tick() { let now = crate::hw::timer::dwt_cyccnt(); let window_start = WINDOW_START_CYCLES.load(Ordering::Relaxed); let window_age = now.wrapping_sub(window_start); if window_age >= crate::config::ISR_QUOTA_WINDOW_CYCLES { // New window: reset counters and unconditionally unmask. ISR_CYCLES_THIS_WINDOW.store(0, Ordering::Relaxed); WINDOW_START_CYCLES.store(now, Ordering::Relaxed); if NON_CRITICAL_MASKED.swap(false, Ordering::Relaxed) { unsafe { unmask_non_critical_irqs(); } defmt::debug!("[IRQ] Quota window reset — non-critical IRQs re-enabled."); } return; } // Still within the window — check if quota is exceeded. let isr_cycles = ISR_CYCLES_THIS_WINDOW.load(Ordering::Relaxed); if isr_cycles > crate::config::ISR_QUOTA_MAX_CYCLES && !NON_CRITICAL_MASKED.load(Ordering::Relaxed) { NON_CRITICAL_MASKED.store(true, Ordering::Relaxed); unsafe { mask_non_critical_irqs(); } defmt::warn!( "[IRQ] Quota exceeded ({} / {} cycles) — masking non-critical IRQs.", isr_cycles, crate::config::ISR_QUOTA_MAX_CYCLES ); } } // ── Platform-specific NVIC helpers ─────────────────────────────────────────── /// Mask UART, Ethernet, and other non-timing-critical interrupt lines. /// /// # Safety /// Modifies NVIC ICER registers; must run in privileged mode. unsafe fn mask_non_critical_irqs() { // Platform-specific: clear-enable bits in NVIC_ICER for UART/ETH IRQ numbers. // Example for Cortex-M (IRQ numbers are board-specific): // const NVIC_ICER0: *mut u32 = 0xE000_E180 as *mut u32; // core::ptr::write_volatile(NVIC_ICER0, UART_IRQ_BIT | ETH_IRQ_BIT); } /// Re-enable previously masked non-critical interrupt lines. /// /// # Safety /// Modifies NVIC ISER registers; must run in privileged mode. unsafe fn unmask_non_critical_irqs() { // Platform-specific: set-enable bits in NVIC_ISER for UART/ETH IRQ numbers. // const NVIC_ISER0: *mut u32 = 0xE000_E100 as *mut u32; // core::ptr::write_volatile(NVIC_ISER0, UART_IRQ_BIT | ETH_IRQ_BIT); }

Key Invariants

MPU Region 0 is always the Kernel Stack Guard — never reassigned.

RIGHT_PIN is always set before ipc_grant and cleared by ipc_revoke.

mark_start / mark_end are called only from the scheduler — no atomics required.

quota_tick and watchdog_tick are called once per scheduler tick in that order.

handle_fault(KernelCore) never returns.
