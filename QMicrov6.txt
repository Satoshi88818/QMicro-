// ── [IMP-7] Deferred revocation queue ──────────────────────────────────────── // // `OwnedCap::drop` pushes here instead of locking the cap table directly. // The scheduler drains this during the idle/telemetry pass (safe moment, // no cap-table lock is held by any caller at that point). static DEFERRED_REVOKE: IrqSafeMutex<Queue<CapToken, DEFERRED_REVOKE_QUEUE_SIZE>> = IrqSafeMutex::new(Queue::new()); } // ── First Class Capabilities (RAII) ────────────────────────────────────────── /// An owned capability guard. immediate revoke for slot {}", self.token.index ); drop(q); // release queue lock before acquiring cap table lock let _ = GLOBAL_CAP_TABLE.lock().revoke_token(self.token); } else { defmt::trace!("Queued deferred revoke: slot {}", self.token.index); } } } // ── Internal Capability Table ───────────────────────────────────────────────── #[derive(Clone, Copy)] pub struct Capability { pub owner: Pid, pub domain: QuantumDomain, pub epoch: Epoch, pub rights: u8, } #[derive(Clone, Copy)] enum SlotState { Free { next: u16 }, Occupied(Capability), } #[derive(Clone, Copy)] struct CapSlot { gen: u16, state: SlotState, } pub struct CapabilityTable { slots: [CapSlot; CAP_TABLE_SIZE], free_head: u16, } impl CapabilityTable { pub const fn new() -> Self { let mut slots = [CapSlot { gen: 1, state: SlotState::Free { next: 0 } }; CAP_TABLE_SIZE]; let mut i = 0; while i < CAP_TABLE_SIZE - 1 { slots[i].state = SlotState::Free { next: (i + 1) as u16 }; i += 1; } slots[CAP_TABLE_SIZE - 1].state = SlotState::Free { next: u16::MAX }; Self { slots, free_head: 0 } } pub fn mint(&mut self, cap: Capability) -> KResult<CapToken> { let idx = self.free_head; if idx == u16::MAX { return Err(KernelError::CapTableFull); } let slot = &mut self.slots[idx as usize]; if let SlotState::Free { next } = slot.state { self.free_head = next; } else { unreachable!(); // Structural integrity failure } slot.gen = slot.gen.wrapping_add(1) | 1; // Odd gen == occupied slot.state = SlotState::Occupied(cap); Ok(CapToken { index: idx, gen: slot.gen }) } /// Look up a capability token. /// /// # [IMP-4] Lock-free epoch check /// Reads `DOMAIN_EPOCHS[domain]` with `Ordering::Acquire` — no mutex needed. pub fn lookup(&self, token: CapToken, required_rights: u8) -> KResult<Capability> { let idx = token.index as usize; if idx >= CAP_TABLE_SIZE { return Err(KernelError::CapOutOfRange); } let slot = &self.slots[idx]; if slot.gen != token.gen { return Err(KernelError::CapStaleGen); } match slot.state { SlotState::Occupied(cap) => { if cap.rights & required_rights != required_rights { return Err(KernelError::PermissionDenied); } // [IMP-4] Acquire load — lock-free, no mutex. let domain_epoch = DOMAIN_EPOCHS[cap.domain.0 as usize].load(Ordering::Acquire); if cap.epoch.0 < domain_epoch { return Err(KernelError::CapRevoked); } Ok(cap) } SlotState::Free { .. } => Err(KernelError::CapEmpty), } } pub fn revoke_token(&mut self, token: CapToken) -> KResult<()> { let idx = token.index as usize; if idx >= CAP_TABLE_SIZE { return Err(KernelError::CapOutOfRange); } let slot = &mut self.slots[idx]; if slot.gen != token.gen { return Err(KernelError::CapStaleGen); } match slot.state { SlotState::Occupied(_) => { slot.gen = slot.gen.wrapping_add(1) & !1; // Even gen == free slot.state = SlotState::Free { next: self.free_head }; self.free_head = idx as u16; Ok(()) } SlotState::Free { .. } => Err(KernelError::CapEmpty), } } }

//! IPC: Message Queues & Cycle-Accurate Grants. //! //! # Changes from v5 //! - [IMP-3] Per-domain receiver queue mutexes replace the single global array mutex. //! Domain A sending to Domain B no longer blocks Domain C from reading its queue. //! - [IMP-2] `check_overlap` replaced by a fixed MPU slot matrix — O(1) slot allocation //! per domain. Grants are partitioned into `MPU_SLOTS_PER_DOMAIN` slots per domain; //! assigning a new grant just finds a free slot in the target domain's row. //! - [IMP-6] `sys_send` and `map_typed_grant` now require `&OwnedCap` instead of raw //! `CapToken`, forcing callers into the safe RAII lifecycle. //! - [IMP-8] `ACTIVE_GRANTS` now tracks `source_domain` as well as `target_domain`. //! `revoke_grants_for_domain` revokes both directions. #![deny(unsafe_code)] use qmicro_schema::{Pid, Message, TypedGrant, QuantumDomain}; use crate::error::{KResult, KernelError}; use crate::capability::{GLOBAL_CAP_TABLE, OwnedCap}; use qmicro_schema::{RIGHT_MAP, RIGHT_INVOKE}; use crate::sync::IrqSafeMutex; use crate::config::{IPC_GRANT_SLOTS, NUM_DOMAINS, MSG_QUEUE_SIZE, MPU_SLOTS_PER_DOMAIN}; use heapless::Vec; // ── [IMP-3] Per-domain receiver queue mutexes ───────────────────────────────── // // Previously: one `IrqSafeMutex<[MsgQueue; NUM_DOMAINS]>` — a single lock for all // domains. Now: each domain has its own independent lock so parallel sends to // different domains are fully concurrent. pub struct MsgQueue { buffer: [Message; MSG_QUEUE_SIZE], head: usize, tail: usize, count: usize, } impl MsgQueue { pub const fn new() -> Self { Self { buffer: [Message { label: 0, args: [0; 4], capability: None, reply_cap: None }; MSG_QUEUE_SIZE], head: 0, tail: 0, count: 0, } } pub fn push(&mut self, msg: Message) -> KResult<()> { if self.count == MSG_QUEUE_SIZE { return Err(KernelError::IpcReceiverQueueFull); } self.buffer[self.tail] = msg; self.tail = (self.tail + 1) % MSG_QUEUE_SIZE; self.count += 1; Ok(()) } pub fn pop(&mut self) -> KResult<Message> { if self.count == 0 { return Err(KernelError::IpcReceiverQueueEmpty); } let msg = self.buffer[self.head]; self.head = (self.head + 1) % MSG_QUEUE_SIZE; self.count -= 1; Ok(msg) } pub fn clear(&mut self) { self.head = 0; self.tail = 0; self.count = 0; } } // [IMP-3] One mutex per domain queue — fine-grained locking. static RECEIVER_QUEUES: [IrqSafeMutex<MsgQueue>; NUM_DOMAINS] = { const INIT: IrqSafeMutex<MsgQueue> = IrqSafeMutex::new(MsgQueue::new()); [INIT; NUM_DOMAINS] }; pub fn clear_queue(domain: QuantumDomain) { RECEIVER_QUEUES[domain.0 as usize].lock().clear(); } // ── [IMP-2] O(1) MPU Grant Slot Matrix ─────────────────────────────────────── // // Instead of an O(N) linear scan over all active grants, each domain has a fixed // row of `MPU_SLOTS_PER_DOMAIN` slots. Assigning a new grant is O(MPU_SLOTS_PER_DOMAIN) // which is a small compile-time constant (4 in config.rs) — effectively O(1). // // [IMP-8] Each slot stores `(source_domain, target_domain, TypedGrant)` to support // bi-directional revocation. #[derive(Clone, Copy)] enum GrantSlot { Empty, /// `(source_domain, grant)` Occupied(QuantumDomain, TypedGrant), } /// Grant table indexed by `[target_domain][slot_index]`. static ACTIVE_GRANTS: IrqSafeMutex< [[GrantSlot; MPU_SLOTS_PER_DOMAIN]; NUM_DOMAINS], > = IrqSafeMutex::new([[GrantSlot::Empty; MPU_SLOTS_PER_DOMAIN]; NUM_DOMAINS]); // ── L4 Message Passing ──────────────────────────────────────────────────────── /// Send a message to `target_domain`. /// /// # [IMP-6] Requires `&OwnedCap` /// Callers must hold an `OwnedCap`, guaranteeing the capability goes through /// the proper RAII lifecycle and cannot be silently aliased. pub fn sys_send( target_domain: QuantumDomain, cap: &OwnedCap, // [IMP-6] msg: Message, ) -> KResult<()> { { let table = GLOBAL_CAP_TABLE.lock(); let _cap = table.lookup(cap.token, RIGHT_INVOKE)?; } // cap table lock released before touching the queue // [IMP-3] Only lock the target domain's queue, not all queues. RECEIVER_QUEUES[target_domain.0 as usize].lock().push(msg)?; defmt::trace!("IPC Send to Domain {}: label={}", target_domain.0, msg.label); Ok(()) } pub fn sys_recv(my_domain: QuantumDomain) -> KResult<Message> { // [IMP-3] Only locks this domain's queue. RECEIVER_QUEUES[my_domain.0 as usize].lock().pop() } // ── MPU Memory Grants ───────────────────────────────────────────────────────── /// Map a typed memory grant into `target_domain`'s MPU. /// /// # [IMP-6] Requires `&OwnedCap` /// # [IMP-2] O(1) slot allocation within `target_domain`'s fixed row /// # [IMP-8] Records `source_domain` for bi-directional revocation pub fn map_typed_grant( source: Pid, source_domain: QuantumDomain, // [IMP-8] new parameter target_domain: QuantumDomain, cap: &OwnedCap, // [IMP-6] grant: TypedGrant, ) -> KResult<()> { { let table = GLOBAL_CAP_TABLE.lock(); let resolved = table.lookup(cap.token, RIGHT_MAP)?; if resolved.owner != source { return Err(KernelError::CapRevoked); } } // cap table lock released let (addr, size, read_only) = get_grant_details(&grant); // [IMP-2] Find a free slot in target_domain's row — O(MPU_SLOTS_PER_DOMAIN). { let mut grants = ACTIVE_GRANTS.lock(); let row = &mut grants[target_domain.0 as usize]; let slot = row .iter_mut() .find(|s| matches!(s, GrantSlot::Empty)) .ok_or(KernelError::IpcGrantSlotFull)?; *slot = GrantSlot::Occupied(source_domain, grant); // [IMP-8] store source } crate::hw::mpu::configure_grant_region(target_domain, addr, size, read_only)?; Ok(()) } /// Revoke all grants involving `target` as either source **or** destination. /// /// # [IMP-8] Bi-directional revocation /// Previously only removed entries where the target domain was the *receiver*. /// Now also clears entries where the target domain was the *source/owner*, /// preventing healthy domains from holding dangling mappings into crashed memory. pub fn revoke_grants_for_domain(target: QuantumDomain) { let mut grants = ACTIVE_GRANTS.lock(); for domain_idx in 0..NUM_DOMAINS { let row = &mut grants[domain_idx]; for slot in row.iter_mut() { let should_clear = match slot { // Grant mapped *into* the target domain GrantSlot::Occupied(_, _) if domain_idx == target.0 as usize => true, // Grant *sourced from* the target domain mapped into a healthy domain // [IMP-8] GrantSlot::Occupied(src, _) if src.0 == target.0 => true, _ => false, }; if should_clear { *slot = GrantSlot::Empty; let receiver = QuantumDomain(domain_idx as u8); let _ = crate::hw::mpu::unmap_grant_region(receiver); defmt::warn!( "Revoked grant: domain {} (target={})", domain_idx, target.0 ); } } } } fn get_grant_details(grant: &TypedGrant) -> (usize, usize, bool) { match grant { TypedGrant::QuantumPulseBuffer(g, _) => (g.addr, g.size, g.read_only), TypedGrant::ClassicalControl(g) => (g.addr, g.size, g.read_only), TypedGrant::MeasurementResult(g) => (g.addr, g.size, true), } }

//! EDF Scheduler. //! //! # Changes from v5 //! - [IMP-1] `check_stack_watermark` removed from the hot `tick()` path. //! It now lives in `telemetry_idle_pass()`, called only when the run queue //! is empty (i.e., during true idle time). /// Task Creation API with safe stack initialization. /// /// # [IMP-5] Safe stack ownership /// Accepts `&'static mut [u32]` instead of raw `(usize, usize)`. /// - The `'static` lifetime ensures the memory outlives any task. /// - `&mut` guarantees exclusive ownership — no aliasing is possible. /// - Filling with `STACK_SENTINEL` uses safe slice methods; no `unsafe` block needed. pub fn spawn_task( pid: Pid, domain: QuantumDomain, deadline: Nanoseconds, stack: &'static mut [u32], // [IMP-5] ) -> KResult<()> { // Safe: `fill` is a plain slice method. stack.fill(STACK_SENTINEL); let stack_base = stack.as_ptr() as usize; let stack_size = stack.len() * 4; // words → bytes let task = Task { deadline, pid, domain, stack_base, stack_size, state: TaskState::Ready, }; RUN_QUEUE.lock().push(task).map_err(|_| KernelError::RunQueueFull) } /// Main scheduler tick — called from the top-level interrupt or timer handler. /// /// # Critical path obligations /// - MUST complete in bounded time (no O(N) scans). /// - [IMP-1] Stack watermark scan is NOT called here. /// - [IMP-9] Overrunning tasks are aborted (popped and dropped) before dispatch. pub fn tick() -> KResult<()> { watchdog::pet(); // Handle domain recovery requests signalled from interrupt context. let bitmap = PREEMPT_BITMAP.swap(0, Ordering::SeqCst); if bitmap != 0 { watchdog::handle_domain_recovery(bitmap); } crate::hw::p4::drain_ring(); let now = timer::now_ns(); let mut queue = RUN_QUEUE.lock(); let pending_task = queue .peek() .filter(|t| t.state == TaskState::Ready) .copied(); if let Some(task) = pending_task { if task.deadline <= now { // [IMP-9] Check overrun BEFORE popping so we can abort without dispatch. let overrun_ns = now.0.saturating_sub(task.deadline.0); watchdog::handle_overrun(task.domain); // Signal recovery return Ok(()); // Do NOT fall through to dispatch } // Deadline met within tolerance — dispatch normally. queue.pop(); drop(queue); // Hardware MPU stack guard is always active — prevents actual overflow. // [IMP-1] Software watermark scan moved to telemetry_idle_pass(). unsafe { mpu::configure_stack_guard(task.stack_base, task.stack_size); } dispatch_task(task); } else { // Task deadline is in the future; arm one-shot timer and sleep. let delay = Nanoseconds(task.deadline.0.saturating_sub(now.0)); drop(queue); unsafe { timer::arm_one_shot(delay); } hw_sleep(); } } else { // Run queue is empty — safe to perform background maintenance. drop(queue); telemetry_idle_pass(); hw_sleep(); } Ok(()) } /// Background maintenance executed only during true idle (run queue empty). /// /// # [IMP-1] Stack watermark scan moved here from tick() /// This eliminates O(N) jitter from the EDF hot path.
