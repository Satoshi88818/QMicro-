### `qmicro/src/capability.rs`
```rust
//! Unified Capability Table.
//! - Enforces capability `rights` (READ, WRITE, DELEGATE, MAP).
//! - Replaces the unsafe global pointer with an `IrqSafeMutex`.
//! - Generation counters prevent double-free / stale handle bugs.

use crate::sync::IrqSafeMutex;
use crate::config::{NUM_DOMAINS, CAP_TABLE_SIZE};
use crate::error::{KernelError, KResult};
use qmicro_schema::{Pid, Epoch, CapToken};

static DOMAIN_EPOCHS: IrqSafeMutex<[Epoch; NUM_DOMAINS]> = IrqSafeMutex::new([Epoch(0); NUM_DOMAINS]);

pub static GLOBAL_CAP_TABLE: IrqSafeMutex<CapabilityTable> = IrqSafeMutex::new(CapabilityTable::new());

#[derive(Clone, Copy)]
struct CapSlot {
    gen: u16,
    cap: Option<Capability>,
}

#[derive(Clone, Copy)]
pub struct Capability {
    pub owner: Pid,
    pub domain_id: u8,
    pub epoch: Epoch,
    pub rights: u8, // Bitmask (RIGHT_READ, RIGHT_WRITE, etc.)
}

pub struct CapabilityTable {
    slots: [CapSlot; CAP_TABLE_SIZE],
    free_head: u16,
}

impl CapabilityTable {
    pub const fn new() -> Self {
        let mut slots = [CapSlot { gen: 0, cap: None }; CAP_TABLE_SIZE];
        let mut i = 0;
        while i < CAP_TABLE_SIZE - 1 {
            slots[i].gen = (i as u16 + 1); 
            i += 1;
        }
        slots[CAP_TABLE_SIZE - 1].gen = u16::MAX;
        Self { slots, free_head: 0 }
    }

    pub fn mint(&mut self, cap: Capability) -> KResult<CapToken> {
        let idx = self.free_head;
        if idx == u16::MAX {
            return Err(KernelError::CapTableFull);
        }
        let slot = &mut self.slots[idx as usize];
        self.free_head = slot.gen; 

        // Advance generation, mark odd (occupied)
        slot.gen = slot.gen.wrapping_add(1) | 1;
        slot.cap = Some(cap);

        Ok(CapToken { index: idx, gen: slot.gen })
    }

    /// Lookup a capability and strictly enforce rights and epoch validity.
    pub fn lookup(&self, token: CapToken, required_rights: u8) -> KResult<Capability> {
        let idx = token.index as usize;
        if idx >= CAP_TABLE_SIZE { return Err(KernelError::CapOutOfRange); }
        
        let slot = &self.slots[idx];
        if slot.gen != token.gen { return Err(KernelError::CapStaleGen); }
        
        let cap = slot.cap.ok_or(KernelError::CapEmpty)?;
        
        // Enforce rights bitmask
        if cap.rights & required_rights != required_rights {
            return Err(KernelError::PermissionDenied);
        }

        // Constant-time epoch validation
        let epochs = DOMAIN_EPOCHS.lock();
        let domain_epoch = epochs[cap.domain_id as usize];
        if cap.epoch.0.wrapping_add(1) <= domain_epoch.0 {
            return Err(KernelError::CapRevoked);
        }

        Ok(cap)
    }

    pub fn revoke_token(&mut self, token: CapToken) -> KResult<()> {
        let idx = token.index as usize;
        if idx >= CAP_TABLE_SIZE { return Err(KernelError::CapOutOfRange); }
        
        let slot = &mut self.slots[idx];
        if slot.gen != token.gen { return Err(KernelError::CapStaleGen); }
        
        slot.cap = None;
        slot.gen = (slot.gen.wrapping_add(1) & !1); // Even = free
        slot.gen = self.free_head;
        self.free_head = idx as u16;
        
        // TODO: Trigger IPC hook to invalidate derived memory grants
        Ok(())
    }
}
```

### `qmicro/src/scheduler.rs`
```rust
//! Scheduler with Deadline Overrun Detection, Stack Watermarking, and Idle WFI.

use core::sync::atomic::{AtomicU32, Ordering};
use crate::config::{NUM_DOMAINS, OVERRUN_THRESHOLD_NS, STACK_SENTINEL};
use crate::hw::{timer, watchdog, mpu};
use crate::sync::IrqSafeMutex;
use heapless::binary_heap::{BinaryHeap, Min};
use qmicro_schema::{Nanoseconds, Pid};

static PREEMPT_BITMAP: AtomicU32 = AtomicU32::new(0);

pub fn signal_preemption(domain_id: u8) {
    PREEMPT_BITMAP.fetch_or(1 << domain_id, Ordering::SeqCst);
}

#[derive(Clone, Copy, PartialEq, Eq)]
pub struct Task {
    pub deadline: Nanoseconds,
    pub pid: Pid,
    pub domain_id: u8,
    pub stack_base: usize,
    pub stack_size: usize,
}

impl PartialOrd for Task {
    fn partial_cmp(&self, other: &Self) -> Option<core::cmp::Ordering> {
        Some(self.deadline.cmp(&other.deadline))
    }
}
impl Ord for Task {
    fn cmp(&self, other: &Self) -> core::cmp::Ordering {
        self.deadline.cmp(&other.deadline)
    }
}

static RUN_QUEUE: IrqSafeMutex<BinaryHeap<Task, Min, 32>> = IrqSafeMutex::new(BinaryHeap::new());

pub fn tick() {
    watchdog::pet();

    // 1. Service high-priority preemptions/revocations
    let bitmap = PREEMPT_BITMAP.swap(0, Ordering::SeqCst);
    if bitmap != 0 {
        // Promote recovery task or handle immediately
        watchdog::handle_domain_recovery(bitmap);
    }

    crate::hw::p4::drain_ring();

    let now = timer::now_ns();
    let mut queue = RUN_QUEUE.lock();

    if let Some(task) = queue.peek() {
        if task.deadline <= now {
            // Overrun Detection
            if now.0.saturating_sub(task.deadline.0) > OVERRUN_THRESHOLD_NS {
                watchdog::handle_overrun(task.domain_id);
            }

            let task = *task;
            queue.pop();
            drop(queue);

            // Setup memory protection
            unsafe {
                mpu::configure_stack_guard(task.stack_base, task.stack_size);
                check_stack_watermark(&task);
            }

            dispatch_task(task);
        } else {
            // Tickless sleep
            let delay = Nanoseconds(task.deadline.0.saturating_sub(now.0));
            drop(queue);
            unsafe { timer::arm_one_shot(delay) };
            
            // Go to low-power state
            #[cfg(target_arch = "arm")]
            unsafe { core::arch::asm!("wfi", options(nomem, nostack)); }
        }
    } else {
        drop(queue);
        // Queue empty, wait for interrupt
        #[cfg(target_arch = "arm")]
        unsafe { core::arch::asm!("wfi", options(nomem, nostack)); }
    }
}

unsafe fn check_stack_watermark(task: &Task) {
    // Check if the bottom of the stack (above redzone) still contains the sentinel
    let sentinel_ptr = (task.stack_base + crate::config::STACK_REDZONE_BYTES) as *const u32;
    if core::ptr::read_volatile(sentinel_ptr) != STACK_SENTINEL {
        watchdog::trigger_panic(true, Some(task.domain_id), "Stack near-overflow detected!");
    }
}

fn dispatch_task(_task: Task) {}
```

// ── MPU Memory Grants ─────────────────────────────────────────────────────────

pub fn map_typed_grant(source: Pid, target: Pid, token: CapToken, grant: TypedGrant) -> KResult<()> {
    let table = GLOBAL_CAP_TABLE.lock();
    let cap = table.lookup(token, RIGHT_MAP)?;
    if cap.owner != source { return Err(KernelError::CapRevoked); }

    let mut active = ACTIVE_GRANTS.lock();
    check_overlap(&active, &grant)?;

    // Insert active grant for later revocation tracking
    let _ = active.push(grant);

    // Hardware MPU programming based on grant type (Quantum / Classical / Result)
    let (addr, size, read_only) = match grant {
        TypedGrant::QuantumPulseBuffer(g) => (g.addr, g.size, g.read_only),
        TypedGrant::ClassicalControl(g) => (g.addr, g.size, g.read_only),
        TypedGrant::MeasurementResult(g) => (g.addr, g.size, true), // Forced read-only
    };

    unsafe { crate::hw::mpu::configure_grant_region(target, addr, size, read_only)?; }
    Ok(())
}

fn check_overlap(active: &Vec<TypedGrant, IPC_GRANT_SLOTS>, new: &TypedGrant) -> KResult<()> {
    let get_range = |tg: &TypedGrant| match tg {
        TypedGrant::QuantumPulseBuffer(g) | TypedGrant::ClassicalControl(g) | TypedGrant::MeasurementResult(g) 
            => (g.addr, g.addr + g.size),
    };

    let (n_start, n_end) = get_range(new);
    for existing in active.iter() {
        let (e_start, e_end) = get_range(existing);
        // Standard overlap interval check
        if n_start < e_end && e_start < n_end {
            return Err(KernelError::IpcGrantOverlap);
        }
    }
    Ok(())
}

//! Deterministic Fixed-Point Math.
//! - CORDIC for sin/cos/atan2
//! - Saturating Arithmetic
//! - No Floating Point (`f64`/`f32` strictly forbidden)

use crate::error::{KResult, KernelError};

pub type Q0_32 = u32;

/// CORDIC implementation for simultaneous Sine and Cosine.
/// Inputs are scaled to `0..u32::MAX` representing `0..2π`.
pub fn cordic_sin_cos(angle: u32) -> (i32, i32) {
    // Precomputed arctan table for CORDIC iterations
    const ATAN_TABLE: [u32; 16] = [
        0x20000000, 0x12E4051D, 0x09D8839D, 0x04FA9BE9, 
        0x027FA201, 0x013FF28F, 0x009FFC90, 0x004FFFF6,
        0x0027FFF5, 0x0013FFFE, 0x000A0000, 0x00050000,
        0x00028000, 0x00014000, 0x0000A000, 0x00005000,
    ];
    const K: i32 = 0x26DD3B6A; // Scaling factor ~0.607252

    let mut x: i32 = K;
    let mut y: i32 = 0;
    
    // Map angle from full circle to [-π/2, π/2] for convergence
    let mut z = (angle as i32) ^ 0x80000000; 

    for i in 0..16 {
        let d = if z >= 0 { 1 } else { -1 };
        let tx = x - d * (y >> i);
        let ty = y + d * (x >> i);
        let tz = z - d * (ATAN_TABLE[i] as i32);
        
        x = tx;
        y = ty;
        z = tz;
    }
    
    // Return (sin, cos)
    (y, x)
}

/// 2x2 Fixed point complex matrix multiply with saturating accumulation
pub fn mat2x2_mul_saturating(
    a: [[i32; 2]; 2], 
    b: [[i32; 2]; 2]
) -> [[i32; 2]; 2] {
    let mut c = [[0i32; 2]; 2];
    for i in 0..2 {
        for j in 0..2 {
            let p1 = (a[i][0] as i64 * b[0][j] as i64) >> 16;
            let p2 = (a[i][1] as i64 * b[1][j] as i64) >> 16;
            // Saturating add
            c[i][j] = (p1.saturating_add(p2)) as i32;
        }
    }
    c
}

    #[cfg(target_arch = "arm")]
    unsafe {
        // Read CYCCNT. Note: Handle 32-bit wrap around in actual driver

//! # QMicro v3 Entry Point
//! Initialises hardware, runs the EDF scheduler, and performs minimal panic dumps.

#![no_std]
#![no_main]

pub mod config;
pub mod error;
pub mod sync;
pub mod capability;
pub mod scheduler;
pub mod ipc;
pub mod math;
pub mod hw {
    pub mod interrupts;
    pub mod mpu;
    pub mod timer;
    pub mod p4;
    pub mod watchdog;
}

#[derive(Default)]
struct RegisterDump {
    r0: u32, r1: u32, r2: u32, r3: u32,
    pc: u32, lr: u32,
}

#[panic_handler]
fn panic_handler(info: &core::panic::PanicInfo) -> ! {
    // Minimal Register Dump for debug bring-up
    let mut regs = RegisterDump::default();
    
    #[cfg(target_arch = "arm")]
    unsafe {
        core::arch::asm!(
            "mov {0}, r0", "mov {1}, r1", "mov {2}, r2", "mov {3}, r3",
            "mov {4}, pc", "mov {5}, lr",
            out(reg) regs.r0, out(reg) regs.r1, out(reg) regs.r2, out(reg) regs.r3,
            out(reg) regs.pc, out(reg) regs.lr
        );
    }

    // Zero-cost defmt log (stubbed for non-hardware targets)
    // defmt::error!("Kernel Panic: {} | PC: {=u32:X}", info, regs.pc);

    hw::watchdog::trigger_panic(false, None, "Critical System Panic");
}

#[no_mangle]
pub extern "C" fn _start() -> ! {
    hw::watchdog::init();
    unsafe { hw::mpu::configure_code_data_regions(); }

    loop {
        scheduler::tick();
    }
}

//! Formal Verification Models

#[cfg(kani)]
mod proofs {
    use super::*;
    use crate::capability::{CapabilityTable, Capability, RIGHT_READ, RIGHT_INVOKE};
    use crate::math::cordic_sin_cos;
    use qmicro_schema::{Pid, Epoch};

    // ── Math: CORDIC never panics/overflows ─────────────────────────────────
    #[kani::proof]
    fn prove_cordic_no_panic() {
        let angle: u32 = kani::any();
        let (sin, cos) = cordic_sin_cos(angle);
        // Verify bounding limits [-K, K] mapped to domain
        assert!(sin >= -0x40000000 && sin <= 0x40000000);
    }

    // ── Capability: Lifecycle and Rights Enforcement ────────────────────────
    #[kani::proof]
    fn prove_capability_lifecycle() {
        let mut table = CapabilityTable::new();
        
        let cap = Capability {
            owner: Pid(1),
            domain_id: 0,
            epoch: Epoch(1),
            rights: RIGHT_READ,
        };

        let token = table.mint(cap).unwrap();

        // 1. Should succeed with READ rights
        assert!(table.lookup(token, RIGHT_READ).is_ok());

        // 2. Should fail if requiring INVOKE rights
        assert!(table.lookup(token, RIGHT_INVOKE).is_err());

        // 3. Revoke token
        assert!(table.revoke_token(token).is_ok());

        // 4. Stale lookup MUST fail
        assert!(table.lookup(token, RIGHT_READ).is_err());
    }

    // ── Scheduler: EDF Monotonicity ─────────────────────────────────────────
    #[kani::proof]
    fn prove_edf_monotonicity() {
        use heapless::binary_heap::{BinaryHeap, Min};
        use crate::scheduler::Task;
        use qmicro_schema::Nanoseconds;

        let mut heap: BinaryHeap<Task, Min, 4> = BinaryHeap::new();
        
        let t1 = Task { deadline: Nanoseconds(100), pid: Pid(1), domain_id: 0, stack_base: 0, stack_size: 1024 };
        let t2 = Task { deadline: Nanoseconds(50), pid: Pid(2), domain_id: 0, stack_base: 0, stack_size: 1024 };
        
        heap.push(t1).unwrap();
        heap.push(t2).unwrap();

        // Must pop lowest deadline first
        let first = heap.pop().unwrap();
        assert_eq!(first.deadline.0, 50);
    }
}
