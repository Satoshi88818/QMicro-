### `qmicro/src/capability.rs`
```rust
//! Unified Capability Table with First-Class RAII.
//! Provides static protections against invalid capabilities.

#![deny(unsafe_code)]

use crate::sync::IrqSafeMutex;
use crate::config::{NUM_DOMAINS, CAP_TABLE_SIZE};
use crate::error::{KernelError, KResult};
use qmicro_schema::{Pid, Epoch, CapToken, QuantumDomain};

static DOMAIN_EPOCHS: IrqSafeMutex<[Epoch; NUM_DOMAINS]> = IrqSafeMutex::new([Epoch(0); NUM_DOMAINS]);

pub static GLOBAL_CAP_TABLE: IrqSafeMutex<CapabilityTable> = IrqSafeMutex::new(CapabilityTable::new());

// ── First Class Capabilities (RAII) ───────────────────────────────────────────

/// An owned capability guard. Enforces RAII drop-revocation.
pub struct OwnedCap {
    pub token: CapToken,
    pub rights_mask: u8,
}

impl Drop for OwnedCap {
    fn drop(&mut self) {
        // Automatically revoke capability when the handle goes out of scope
        let _ = GLOBAL_CAP_TABLE.lock().revoke_token(self.token);
        defmt::trace!("RAII Drop: Revoked Capability {}", self.token.index);
    }
}

// ── Internal Capability Table ─────────────────────────────────────────────────

#[derive(Clone, Copy)]
struct CapSlot {
    gen: u16,
    cap: Option<Capability>,
}

#[derive(Clone, Copy)]
pub struct Capability {
    pub owner: Pid,
    pub domain: QuantumDomain,
    pub epoch: Epoch,
    pub rights: u8,
}

pub struct CapabilityTable {
    slots: [CapSlot; CAP_TABLE_SIZE],
    free_head: u16,
}

impl CapabilityTable {
    pub const fn new() -> Self {
        let mut slots = [CapSlot { gen: 0, cap: None }; CAP_TABLE_SIZE];
        let mut i = 0;
        while i < CAP_TABLE_SIZE - 1 {
            slots[i].gen = (i as u16 + 1); 
            i += 1;
        }
        slots[CAP_TABLE_SIZE - 1].gen = u16::MAX;
        Self { slots, free_head: 0 }
    }

    pub fn mint(&mut self, cap: Capability) -> KResult<CapToken> {
        let idx = self.free_head;
        if idx == u16::MAX {
            return Err(KernelError::CapTableFull);
        }
        let slot = &mut self.slots[idx as usize];
        self.free_head = slot.gen; 

        slot.gen = slot.gen.wrapping_add(1) | 1; // Mark odd (occupied)
        slot.cap = Some(cap);

        Ok(CapToken { index: idx, gen: slot.gen })
    }

    pub fn lookup(&self, token: CapToken, required_rights: u8) -> KResult<Capability> {
        let idx = token.index as usize;
        if idx >= CAP_TABLE_SIZE { return Err(KernelError::CapOutOfRange); }
        
        let slot = &self.slots[idx];
        if slot.gen != token.gen { return Err(KernelError::CapStaleGen); }
        
        let cap = slot.cap.ok_or(KernelError::CapEmpty)?;
        
        if cap.rights & required_rights != required_rights {
            return Err(KernelError::PermissionDenied);
        }

        let epochs = DOMAIN_EPOCHS.lock();
        let domain_epoch = epochs[cap.domain.0 as usize];
        if cap.epoch.0.wrapping_add(1) <= domain_epoch.0 {
            return Err(KernelError::CapRevoked);
        }

        Ok(cap)
    }

    pub fn revoke_token(&mut self, token: CapToken) -> KResult<()> {
        let idx = token.index as usize;
        if idx >= CAP_TABLE_SIZE { return Err(KernelError::CapOutOfRange); }
        
        let slot = &mut self.slots[idx];
        if slot.gen != token.gen { return Err(KernelError::CapStaleGen); }
        
        slot.cap = None;
        slot.gen = (slot.gen.wrapping_add(1) & !1); // Mark even (free)
        slot.gen = self.free_head;
        self.free_head = idx as u16;
        
        Ok(())
    }
}
```

```rust
//! IPC: Message Queues & Cycle-Accurate Grants.

#![deny(unsafe_code)]

use qmicro_schema::{Pid, CapToken, Message, TypedGrant, QuantumDomain};
use crate::error::{KResult, KernelError};
use crate::capability::{GLOBAL_CAP_TABLE, RIGHT_MAP, RIGHT_INVOKE};
use crate::sync::IrqSafeMutex;
use crate::config::{IPC_GRANT_SLOTS, NUM_DOMAINS, MSG_QUEUE_SIZE};
use heapless::Vec;

static ACTIVE_GRANTS: IrqSafeMutex<Vec<TypedGrant, IPC_GRANT_SLOTS>> = IrqSafeMutex::new(Vec::new());

// ── Ringbuffer Message Queues ─────────────────────────────────────────────────

pub struct MsgQueue {
    buffer: [Message; MSG_QUEUE_SIZE],
    head: usize,
    tail: usize,
    count: usize,
}

impl MsgQueue {
    const fn new() -> Self {
        Self { buffer: [Message { label: 0, args: [0;4], capability: None }; MSG_QUEUE_SIZE], head: 0, tail: 0, count: 0 }
    }
    fn push(&mut self, msg: Message) -> KResult<()> {
        if self.count == MSG_QUEUE_SIZE { return Err(KernelError::IpcReceiverQueueFull); }
        self.buffer[self.tail] = msg;
        self.tail = (self.tail + 1) % MSG_QUEUE_SIZE;
        self.count += 1;
        Ok(())
    }
    fn pop(&mut self) -> KResult<Message> {
        if self.count == 0 { return Err(KernelError::IpcReceiverQueueEmpty); }
        let msg = self.buffer[self.head];
        self.head = (self.head + 1) % MSG_QUEUE_SIZE;
        self.count -= 1;
        Ok(msg)
    }
}

static RECEIVER_QUEUES: IrqSafeMutex<[MsgQueue; NUM_DOMAINS]> = IrqSafeMutex::new([MsgQueue::new(); NUM_DOMAINS]);

// ── L4 Message Passing ────────────────────────────────────────────────────────

pub fn sys_send(target_domain: QuantumDomain, token: CapToken, msg: Message) -> KResult<()> {
    let table = GLOBAL_CAP_TABLE.lock();
    let _cap = table.lookup(token, RIGHT_INVOKE)?;
    
    let mut queues = RECEIVER_QUEUES.lock();
    queues[target_domain.0 as usize].push(msg)?;
    
    defmt::trace!("IPC Send to Domain {}: {:?}", target_domain.0, msg.label);
    Ok(())
}

/// Blocking `sys_recv`. In an EDF context, if this returns `IpcReceiverQueueEmpty`, 
/// the scheduler should transition the task to a Blocked state.
pub fn sys_recv(my_domain: QuantumDomain) -> KResult<Message> {
    let mut queues = RECEIVER_QUEUES.lock();
    queues[my_domain.0 as usize].pop()
}

// ── MPU Memory Grants ─────────────────────────────────────────────────────────

pub fn map_typed_grant(source: Pid, target_domain: QuantumDomain, token: CapToken, grant: TypedGrant) -> KResult<()> {
    let table = GLOBAL_CAP_TABLE.lock();
    let cap = table.lookup(token, RIGHT_MAP)?;
    if cap.owner != source { return Err(KernelError::CapRevoked); }

    let mut active = ACTIVE_GRANTS.lock();
    check_overlap(&active, &grant)?;
    let _ = active.push(grant);

    // Call out to HW abstraction for MPU boundary mapping
    let (addr, size, read_only) = match grant {
        TypedGrant::QuantumPulseBuffer(g, _timing) => (g.addr, g.size, g.read_only),
        TypedGrant::ClassicalControl(g) => (g.addr, g.size, g.read_only),
        TypedGrant::MeasurementResult(g) => (g.addr, g.size, true), 
    };

    // Note: We delegate the unsafe hardware programming to the hw module
    crate::hw::mpu::configure_grant_region(target_domain, addr, size, read_only)?;
    Ok(())
}

fn check_overlap(active: &Vec<TypedGrant, IPC_GRANT_SLOTS>, new: &TypedGrant) -> KResult<()> {
    let get_range = |tg: &TypedGrant| match tg {
        TypedGrant::QuantumPulseBuffer(g, _) | TypedGrant::ClassicalControl(g) | TypedGrant::MeasurementResult(g) 
            => (g.addr, g.addr + g.size),
    };

    let (n_start, n_end) = get_range(new);
    for existing in active.iter() {
        let (e_start, e_end) = get_range(existing);
        if n_start < e_end && e_start < n_end {
            return Err(KernelError::IpcGrantOverlap);
        }
    }
    Ok(())
}
```

```rust
//! EDF Scheduler with Stack Watermarking and Idle WFI.

use core::sync::atomic::{AtomicU32, Ordering};
use crate::config::{OVERRUN_THRESHOLD_NS, STACK_SENTINEL, STACK_REDZONE_BYTES};
use crate::hw::{timer, watchdog, mpu};
use crate::sync::IrqSafeMutex;
use heapless::binary_heap::{BinaryHeap, Min};
use qmicro_schema::{Nanoseconds, Pid, QuantumDomain};

static PREEMPT_BITMAP: AtomicU32 = AtomicU32::new(0);

pub fn signal_preemption(domain: QuantumDomain) {
    PREEMPT_BITMAP.fetch_or(1 << domain.0, Ordering::SeqCst);
}

#[derive(Clone, Copy, PartialEq, Eq)]
pub enum TaskState {
    Ready,
    BlockedOnIpc,
}

#[derive(Clone, Copy, PartialEq, Eq)]
pub struct Task {
    pub deadline: Nanoseconds,
    pub pid: Pid,
    pub domain: QuantumDomain,
    pub stack_base: usize,
    pub stack_size: usize,
    pub state: TaskState,
}

impl PartialOrd for Task {
    fn partial_cmp(&self, other: &Self) -> Option<core::cmp::Ordering> {
        Some(self.deadline.cmp(&other.deadline))
    }
}
impl Ord for Task {
    fn cmp(&self, other: &Self) -> core::cmp::Ordering {
        self.deadline.cmp(&other.deadline)
    }
}

static RUN_QUEUE: IrqSafeMutex<BinaryHeap<Task, Min, 32>> = IrqSafeMutex::new(BinaryHeap::new());

pub fn tick() {
    watchdog::pet();

    let bitmap = PREEMPT_BITMAP.swap(0, Ordering::SeqCst);
    if bitmap != 0 {
        watchdog::handle_domain_recovery(bitmap);
    }

    crate::hw::p4::drain_ring();

    let now = timer::now_ns();
    let mut queue = RUN_QUEUE.lock();

    // Find the next Ready task
    let mut pending_task = None;
    if let Some(task) = queue.peek() {
        if task.state == TaskState::Ready {
            pending_task = Some(*task);
        }
    }

    if let Some(task) = pending_task {
        if task.deadline <= now {
            if now.0.saturating_sub(task.deadline.0) > OVERRUN_THRESHOLD_NS {
                defmt::error!("Overrun in Domain {}!", task.domain.0);
                watchdog::handle_overrun(task.domain);
            }

            queue.pop();
            drop(queue);

            // Isolate unsafe HW integration bounds
            unsafe { mpu::configure_stack_guard(task.stack_base, task.stack_size); }
            
            // Observability: Runtime stack usage reporting
            check_stack_watermark(&task);
            
            dispatch_task(task);
        } else {
            let delay = Nanoseconds(task.deadline.0.saturating_sub(now.0));
            drop(queue);
            unsafe { timer::arm_one_shot(delay); }
            hw_sleep();
        }
    } else {
        drop(queue);
        hw_sleep();
    }
}

fn hw_sleep() {
    #[cfg(target_arch = "arm")]
    unsafe { core::arch::asm!("wfi", options(nomem, nostack)); }
}

/// Computes actual stack usage by scanning upwards for the sentinel value.
fn check_stack_watermark(task: &Task) {
    let mut ptr = task.stack_base as *const u32;
    let end = (task.stack_base + task.stack_size) as *const u32;
    let mut unused_bytes = 0;

    // Safety: Pointers are verified against task stack limits mapped in MPU.
    unsafe {
        while ptr < end {
            if core::ptr::read_volatile(ptr) == STACK_SENTINEL {
                unused_bytes += 4;
            } else {
                break;
            }
            ptr = ptr.add(1);
        }
    }

    let used_bytes = task.stack_size.saturating_sub(unused_bytes);
    defmt::trace!("Task {} (Domain {}) Stack: {}/{} bytes used", 
                  task.pid.0, task.domain.0, used_bytes, task.stack_size);

    if unused_bytes < STACK_REDZONE_BYTES {
        defmt::error!("Stack near-overflow in Domain {}!", task.domain.0);
        watchdog::trigger_panic(true, Some(task.domain), "Stack Redzone Breached");
    }
}

fn dispatch_task(_task: Task) {}
```

```rust
//! Deterministic Fixed-Point Math. Strictly safe!

#![deny(unsafe_code)]

pub type Q0_32 = u32;

pub fn cordic_sin_cos(angle: u32) -> (i32, i32) {
    const ATAN_TABLE: [u32; 16] = [
        0x20000000, 0x12E4051D, 0x09D8839D, 0x04FA9BE9, 
        0x027FA201, 0x013FF28F, 0x009FFC90, 0x004FFFF6,
        0x0027FFF5, 0x0013FFFE, 0x000A0000, 0x00050000,
        0x00028000, 0x00014000, 0x0000A000, 0x00005000,
    ];
    const K: i32 = 0x26DD3B6A;

    let mut x: i32 = K;
    let mut y: i32 = 0;
    let mut z = (angle as i32) ^ 0x80000000; 

    for i in 0..16 {
        let d = if z >= 0 { 1 } else { -1 };
        let tx = x - d * (y >> i);
        let ty = y + d * (x >> i);
        let tz = z - d * (ATAN_TABLE[i] as i32);
        
        x = tx;
        y = ty;
        z = tz;
    }
    (y, x)
}
```

```rust
//! Formal Verification Models Expanded for v4.

#[cfg(kani)]
mod proofs {
    use super::*;
    use crate::capability::{CapabilityTable, Capability, OwnedCap, RIGHT_READ, RIGHT_INVOKE};
    use crate::ipc::{MsgQueue};
    use crate::math::cordic_sin_cos;
    use qmicro_schema::{Pid, Epoch, QuantumDomain, Message};

    // ── Capability RAII Revocation ──────────────────────────────────────────
    #[kani::proof]
    fn prove_raii_capability_drop() {
        let mut table = CapabilityTable::new();
        let cap = Capability {
            owner: Pid(1),
            domain: QuantumDomain(0),
            epoch: Epoch(1),
            rights: RIGHT_READ,
        };

        let token = table.mint(cap).unwrap();
        
        {
            let _owned = OwnedCap { token, rights_mask: RIGHT_READ };
            // Ensure valid while in scope
            assert!(table.lookup(token, RIGHT_READ).is_ok());
        } // `_owned` dropped here

        // Assert strictly revoked outside of scope
        assert!(table.lookup(token, RIGHT_READ).is_err());
    }

    // ── IPC Message Queue Bounding ──────────────────────────────────────────
    #[kani::proof]
    fn prove_msg_queue_bounds() {
        let mut q = MsgQueue::new();
        let msg = Message::default();
        
        // Ensure no overflow panic, fails gracefully
        for _ in 0..crate::config::MSG_QUEUE_SIZE {
            assert!(q.push(msg).is_ok());
        }
        assert!(q.push(msg).is_err()); // Queue Full

        // Ensure no underflow panic
        for _ in 0..crate::config::MSG_QUEUE_SIZE {
            assert!(q.pop().is_ok());
        }
        assert!(q.pop().is_err()); // Queue Empty
    }
    
    // ── Math Determinism ────────────────────────────────────────────────────
    #[kani::proof]
    fn prove_cordic_bounds() {
        let angle: u32 = kani::any();
        let (sin, cos) = cordic_sin_cos(angle);
        assert!(sin >= -0x40000000 && sin <= 0x40000000);
        assert!(cos >= -0x40000000 && cos <= 0x40000000);
    }
}
```
